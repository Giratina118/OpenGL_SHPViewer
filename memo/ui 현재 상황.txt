// shapefile 로드, 파싱
void CSHPViewerStudyView::LoadShapefile(const std::filesystem::path& shpPath, const std::filesystem::path& shxPath, const std::filesystem::path& dbfPath)
{
	m_layer = Layer();
	m_shpLoader.Parse(shpPath, shxPath, dbfPath, m_layer);
	RefreshMap();
}

// 파일 열리면 화면 갱신
void CSHPViewerStudyView::RefreshMap()
{
	if (m_layer.polygonObjects.empty() && m_layer.polyLineObjects.empty() && m_layer.pointObjects.empty()) return;

	CRect rect;
	GetClientRect(&rect);
	m_camera.InitCamera(m_layer.boundingBox, rect.Width() * (1.0f - uiPanelWidthRate * 2), rect.Height());

	m_renderer.BuildMesh(m_layer);
	m_renderer.RebuildQuadTree(m_layer);

	m_renderer.ReDraw();
}

// 키 입력
void CSHPViewerStudyView::InputKey(float deltaTime)
{
	float moveSpeed = 300.0f * deltaTime;
	float rotateSpeed = 900.0f * deltaTime;

	float moveX = 0.0f;
	float moveY = 0.0f;

	// WASD (로컬 좌표 이동)
	if (m_keyState.keyW) moveY += moveSpeed;
	if (m_keyState.keyS) moveY -= moveSpeed;
	if (m_keyState.keyA) moveX += moveSpeed;
	if (m_keyState.keyD) moveX -= moveSpeed;
	if (moveX != 0.0f || moveY != 0.0f) m_camera.MoveLocal(moveX, moveY);

	// IJKL (월드 좌표 이동)
	if (m_keyState.keyI) moveY += moveSpeed;
	if (m_keyState.keyK) moveY -= moveSpeed;
	if (m_keyState.keyJ) moveX += moveSpeed;
	if (m_keyState.keyL) moveX -= moveSpeed;
	if (moveX != 0.0f || moveY != 0.0f) m_camera.MoveWorld(moveX, moveY);

	// RF (로컬 좌표 줌)
	if (m_keyState.keyR) m_camera.ZoomLocal(1.0);
	if (m_keyState.keyF) m_camera.ZoomLocal(-1.0);

	// UO (월드 좌표 줌)
	if (m_keyState.keyU) m_camera.ZoomWorld(1.0);
	if (m_keyState.keyO) m_camera.ZoomWorld(-1.0);

	// 회전
	float rotX = 0.0f;
	float rotY = 0.0f;
	float rotZ = 0.0f;
	if (m_keyState.keyQ)	 rotZ -= rotateSpeed;
	if (m_keyState.keyE)	 rotZ += rotateSpeed;
	if (m_keyState.keyLeft)  rotX -= rotateSpeed;
	if (m_keyState.keyRight) rotX += rotateSpeed;
	if (m_keyState.keyUp)    rotY -= rotateSpeed;
	if (m_keyState.keyDown)  rotY += rotateSpeed;
	if (rotX != 0.0f || rotY != 0.0f || rotZ != 0.0f) m_camera.RotateLocal(rotX, rotY, rotZ);
}

// 위치 선택 (픽킹)
void CSHPViewerStudyView::PickingPos()
{
	glm::mat4 invVP = glm::inverse(m_camera.GetMatrix());

	CRect rect;
	GetClientRect(&rect);

	// UI 패널 제외한 실제 3D 렌더 영역
	int panelW = (int)(rect.Width() * uiPanelWidthRate);
	int mapW = rect.Width() - panelW * 2;
	int mapH = rect.Height();

	// 3D 영역 기준 마우스 좌표
	float mousePosX = (float)(m_pickingMousePos.x - panelW);
	float mousePosY = (float)(m_pickingMousePos.y);

	// NDC 변환 (Y축 반전 필수)
	float ndcX = (mousePosX / mapW) * 2.0f - 1.0f;
	float ndcY = -((mousePosY / mapH) * 2.0f - 1.0f);

	glm::vec4 nearClip(ndcX, ndcY, -1.0f, 1.0f);
	glm::vec4 farClip(ndcX, ndcY, 1.0f, 1.0f);

	// inverse 변환 후 w로 나누기 (필수)
	glm::vec4 nearWorld = invVP * nearClip;
	glm::vec4 farWorld = invVP * farClip;

	nearWorld /= nearWorld.w;
	farWorld /= farWorld.w;

	glm::dvec3 rayOrigin = glm::dvec3(nearWorld);
	glm::dvec3 rayDir = glm::normalize(glm::dvec3(farWorld) - rayOrigin);

	// z=0 지면과 교차
	if (std::abs(rayDir.z) < 1e-6f) return; // 수평 방향이면 교차 없음

	double t = -rayOrigin.z / rayDir.z;
	if (t < 0.0f) return; // 카메라 뒤쪽

	glm::dvec3 hitPoint = rayOrigin + t * rayDir;

	m_renderer.m_hitPoint = hitPoint;
}

		return;
	}

	// 피킹된 객체 정보 얻기
	// TODO: 폴리곤 데이터만 일단 적용
	if (m_layer.polygonObjects.size() < 1)
		return;

	// TODO: 피킹 아이디 뽑아내는데 성공, 이제 DBF 후 정보출력 차례
	//m_layer.polygonObjects[pickingDataId];
	CString pickingInfo = m_layer.dbfTable.PrintAttribute(pickingDataId);
	m_renderer.SetSelectedObject(pickingDataId, m_uiState);

	m_rightPanel.SetPickingInfo(pickingInfo); // 자동으로 열림
	//m_panelRightBackground.SetWindowText(pickingInfo);
	Invalidate(FALSE);
}

// WM_SIZE 시 호출, 화면 사이즈 변화 시
void CSHPViewerStudyView::OnSize(UINT nType, int screenWidth, int screenHeight)
{
	CView::OnSize(nType, screenWidth, screenHeight);
	if (screenWidth <= 0 || screenHeight <= 0) return; // 최소화 방어

	m_screenWidth = screenWidth;
	m_screenHeight = screenHeight;
	int32_t panelWidth = m_screenWidth * uiPanelWidthRate;
	m_leftPanel.MoveWindow(0, 0, panelWidth, m_screenHeight);
	m_leftPanel.Resize(panelWidth, m_screenHeight);
	m_rightPanel.Resize(m_screenWidth, m_screenHeight, uiPanelWidthRate);

	m_camera.UpdateAspect(screenWidth * (1.0f - uiPanelWidthRate * 2), screenHeight);
	m_renderer.ReDraw();
	m_renderer.Render(m_camera, m_layer, m_uiState);
}

// 마우스 좌클릭 누르기

{
	if (IsInUIPanel(point)) return; // UI 패널 위에서는 카메라가 반응하지 않도록 차단

	if (m_isPickingMode) {

	}
	else if (m_camera.m_isCameraThirdMode) {

		//m_thirdPoint.y = (float)point.y;
		m_pickingMousePos = point;
		PickingPos();
	}

// 마우스 이동
void CSHPViewerStudyView::OnMouseMove(UINT nFlags, CPoint point)
{
	if (!m_isLButtonDragging && !m_isRButtonDragging) return;

	// 화면 픽셀 이동량 → 세계 좌표 이동량으로 변환
	CPoint mouseDelta = point - m_lastMousePos;

	if (m_isLButtonDragging) { // 좌클릭
		if (m_camera.m_isCameraThirdMode) { // 3인칭 이동

			m_camera.transform.position = m_thirdPointCameraPos;
			m_camera.UpdateMatrix();

			m_pickingMousePos = point;
			PickingPos();
			glm::dvec3 deltaPos = m_thirdPoint - m_renderer.m_hitPoint;
			m_renderer.m_hitPoint = m_thirdPoint;

			m_camera.transform.position += deltaPos;
			m_camera.UpdateMatrix();
			//m_camera.MoveThird(m_thirdPoint, point);
		}
		else // 1인칭 이동
			m_camera.MoveLocal(static_cast<float>(mouseDelta.x), static_cast<float>(mouseDelta.y));
	}
	if (m_isRButtonDragging) { // 우클릭
		if (m_camera.m_isCameraThirdMode) // 3인칭 회전
			m_camera.RotateThird(static_cast<float>(mouseDelta.x), static_cast<float>(mouseDelta.y), 0.0f, m_renderer.m_hitPoint);
		else // 1인칭 회전
			m_camera.RotateLocal(static_cast<float>(mouseDelta.x), static_cast<float>(mouseDelta.y), 0.0f);
	}

	m_lastMousePos = point;

	//CView::OnMouseMove(nFlags, point);
}

// 마우스 휠
BOOL CSHPViewerStudyView::OnMouseWheel(UINT nFlags, short zDelta, CPoint point)
{
	// 휠은 화면 좌표가 아닌 스크린 좌표 → 클라이언트 좌표 변환 후 판정
	CPoint clientPt = point;
	ScreenToClient(&clientPt);
	if (IsInUIPanel(clientPt)) return CView::OnMouseWheel(nFlags, zDelta, point);

	// 일반 휠: 줌
	double factor = (zDelta > 0) ? 1.0 : -1.0;

	if (m_camera.m_isCameraThirdMode) {
		m_pickingMousePos = clientPt;
		PickingPos();
		m_camera.ZoomThird(factor, m_renderer.m_hitPoint);
	}
	else
		m_camera.ZoomLocal(factor);

	return TRUE;
}

void CSHPViewerStudyView::OnDropFiles(HDROP hDropInfo)
{
	UINT fileCount = DragQueryFile(hDropInfo, 0xFFFFFFFF, nullptr, 0);

	m_layer = Layer();  // 초기화

	for (UINT i = 0; i < fileCount; i++)
	{
		TCHAR filePath[MAX_PATH];
		DragQueryFile(hDropInfo, i, filePath, MAX_PATH);

		std::filesystem::path shpPath = filePath;
		if (shpPath.extension() != ".shp") continue;

		std::filesystem::path shxPath = shpPath;
		shxPath.replace_extension(".shx");
		if (!std::filesystem::exists(shxPath)) continue;

		std::filesystem::path dbfPath = shpPath;
		dbfPath.replace_extension(".dbf");
		if (!std::filesystem::exists(dbfPath)) continue;

		m_shpLoader.Parse(shpPath, shxPath, dbfPath, m_layer);
	}

	DragFinish(hDropInfo);
	RefreshMap();
}

// SHP 파일 직접 선택
void CSHPViewerStudyView::OnFileOpenShp()
{
	CFileDialog dlg(TRUE, _T("shp"), nullptr,
		OFN_FILEMUSTEXIST | OFN_HIDEREADONLY,
		_T("Shapefile (*.shp)|*.shp||"));

	if (dlg.DoModal() != IDOK) return;

	std::filesystem::path shpPath = dlg.GetPathName().GetString();

	std::filesystem::path shxPath = shpPath;
	shxPath.replace_extension(".shx");
	if (!std::filesystem::exists(shxPath))
	{
		AfxMessageBox(_T("같은 폴더에 .shx 파일이 없습니다."));
		return;
	}

	std::filesystem::path dbfPath = shpPath;
	dbfPath.replace_extension(".dbf");
	if (!std::filesystem::exists(dbfPath))
	{
		AfxMessageBox(_T("같은 폴더에 .dbf 파일이 없습니다."));
		return;
	}

	LoadShapefile(shpPath, shxPath, dbfPath); // shapefile 로드, 파싱
}

// 폴더 선택 -> 안에 있는 shp 전부 읽기
void CSHPViewerStudyView::OnFileOpenFolder()
{
	// Vista 이상 폴더 선택 다이얼로그
	IFileOpenDialog* pDialog = nullptr;
	CoCreateInstance(CLSID_FileOpenDialog, nullptr, CLSCTX_INPROC_SERVER, IID_PPV_ARGS(&pDialog));

	DWORD options;
	pDialog->GetOptions(&options);
	pDialog->SetOptions(options | FOS_PICKFOLDERS);

	if (FAILED(pDialog->Show(m_hWnd))) { pDialog->Release(); return; }

	IShellItem* pItem = nullptr;
	pDialog->GetResult(&pItem);

	PWSTR folderPath = nullptr;
	pItem->GetDisplayName(SIGDN_FILESYSPATH, &folderPath);

	std::filesystem::path folder = folderPath;
	CoTaskMemFree(folderPath);
	pItem->Release();
	pDialog->Release();

	// 폴더 안 .shp 파일 전부 수집
	m_layer = Layer();  // 초기화
	for (const auto& entry : std::filesystem::directory_iterator(folder))
	{
		if (entry.path().extension() != ".shp") continue;

		std::filesystem::path shxPath = entry.path();
		shxPath.replace_extension(".shx");
		if (!std::filesystem::exists(shxPath)) continue;

		std::filesystem::path dbfPath = entry.path();
		dbfPath.replace_extension(".dbf");
		if (!std::filesystem::exists(dbfPath)) continue;

		m_shpLoader.Parse(entry.path(), shxPath, dbfPath, m_layer);
	}

	RefreshMap();
}

void CLeftPanel::Resize(int w, int h)
{
    int mx = w * 0.05, bw = w - mx * 2;
    int bh = h * 0.08, gap = h * 0.01;
    int tabH = 28;
    int pageTop = tabH + 4;
    int pageH = h - pageTop;

    m_tab.MoveWindow(0, 0, w, tabH + pageH);

    // 세 페이지 모두 같은 위치에 겹쳐 배치 (탭이 전환할 때 ShowWindow로 교체)
    CRect pageRect(mx, pageTop + gap, w - mx, pageTop + pageH - gap);
    m_page1.MoveWindow(pageRect);
    m_page2.MoveWindow(pageRect);
    m_page3.MoveWindow(pageRect);

    // 탭1 내부
    m_staticChangeInfo.MoveWindow(0, 0, bw, bh * 2.5);
    m_staticInfo.MoveWindow(0, bh * 2.5 + gap, bw, bh * 2);

    // 탭2 내부
    for (int i = 0; i < 5; i++) {
        CWnd* btns[] = { &m_buttonObjectMBR,&m_buttonNodeMBR,
                        &m_buttonLevelColor,&m_buttonFrustumView,&m_buttonFakeObject };
        btns[i]->MoveWindow(0, i * (bh + gap), bw, bh);
    }
    m_staticSliderInfo.MoveWindow(0, 5 * (bh + gap), bw, bh * 0.6);
    m_sliderViewRange.MoveWindow(0, 5 * (bh + gap) + bh * 0.6 + gap, bw, bh * 0.5);

    // 탭3 내부
    m_buttonPicking.MoveWindow(0, 0, bw, bh);
    m_radioFirstPerson.MoveWindow(0, bh + gap, bw / 2 - gap, bh);
    m_radioThirdPerson.MoveWindow(bw / 2 + gap, bh + gap, bw / 2 - gap, bh);

    // 폰트
    int fs = std::max(10, h / 32);
    m_font.DeleteObject();
    m_font.CreateFont(-fs, 0, 0, 0, FW_NORMAL, FALSE, FALSE, FALSE,
        DEFAULT_CHARSET, OUT_DEFAULT_PRECIS, CLIP_DEFAULT_PRECIS,
        CLEARTYPE_QUALITY, DEFAULT_PITCH | FF_SWISS, _T("Segoe UI"));

    auto applyFont = [&](CWnd& w) { if (w.GetSafeHwnd()) w.SetFont(&m_font); };
    applyFont(m_staticChangeInfo); applyFont(m_staticInfo);
    applyFont(m_buttonObjectMBR);  applyFont(m_buttonNodeMBR);
    applyFont(m_buttonLevelColor); applyFont(m_buttonFrustumView);
    applyFont(m_buttonFakeObject); applyFont(m_staticSliderInfo);
    applyFont(m_buttonPicking);    applyFont(m_radioFirstPerson);
    applyFont(m_radioThirdPerson);
}
